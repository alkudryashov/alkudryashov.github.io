def calculate_profit(indicators_num):
    years_to_invest = buy_price[indicators_num].keys()
    triggers_num = len(years_to_invest)
    profit = []
    for year in years_to_invest:
        spent = buy_price[indicators_num][year][0]
        gain = sell_price_by_year[int(year)]
        profit.append((gain-spent) / spent * 100)
    return list((triggers_num, np.mean(profit), profit))
